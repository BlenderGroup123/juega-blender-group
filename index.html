        function shuffle(array) {

            for (let i = array.length - 1; i > 0; i--) {

                const j = Math.floor(Math.random() * (i + 1));

                [array[i], array[j]] = [array[j], array[i]];

            }

            return array;

        }


        function createBoard() {

            gameBoard.innerHTML = '';

            isBoardLocked = true;

            matchedPairs = 0;

            totalPairs = level + 1;


            const imagePairs = IMAGE_URLS.slice(0, totalPairs);

            let deck = imagePairs.flatMap(url => [{ type: 'image', value: url }, { type: 'image', value: url }]);

            

            let trapTypes = [];

            if (level >= 4) trapTypes.push('octopus');

            if (level >= 6) trapTypes.push('bomb');

            if (level >= 8) trapTypes.push('tornado');


            while (trapTypes.length > 2) {

                trapTypes.splice(Math.floor(Math.random() * trapTypes.length), 1);

            }

            

            trapTypes.forEach(trap => {

                if (trap === 'octopus') deck.push({ type: 'special', value: 'octopus', icon: '🐙' });

                if (trap === 'bomb') deck.push({ type: 'special', value: 'bomb', icon: '💣' }, { type: 'special', value: 'bomb', icon: '💣' });

                if (trap === 'tornado') deck.push({ type: 'special', value: 'tornado', icon: '🌪️' });

            });

            

            const shuffledDeck = shuffle(deck);


            const columns = Math.ceil(Math.sqrt(shuffledDeck.length));

            gameBoard.style.gridTemplateColumns = `repeat(${columns}, 1fr)`;

            

            shuffledDeck.forEach(cardData => {

                const cardElement = document.createElement('div');

                cardElement.classList.add('card-container', 'aspect-square');

                const backFaceContent = cardData.type === 'image'

                    ? `<img src="${cardData.value}" alt="Imagen de carta" onerror="this.onerror=null;this.src='https://placehold.co/100/ccc/fff?text=Error';">`

                    : `<span class="text-4xl">${cardData.icon}</span>`;


                cardElement.innerHTML = `

                    <div class="card" data-type="${cardData.type}" data-value="${cardData.value}">

                        <div class="card-face card-front">🐵</div>

                        <div class="card-face card-back">${backFaceContent}</div>

                    </div>`;

                gameBoard.appendChild(cardElement);

                cardElement.querySelector('.card').addEventListener('click', handleCardClick);

            });

            

            setTimeout(() => { isBoardLocked = false; startTimer(); }, 500);

        }


        function handleCardClick(e) {

            if (isBoardLocked || !isGameActive) return;

            const clickedCard = e.currentTarget;

            if (clickedCard.classList.contains('flipped') || clickedCard.classList.contains('matched')) return;

            

            clickSound.triggerAttackRelease('G3', '16n');

            clickedCard.classList.add('flipped');


            const cardType = clickedCard.dataset.type;

            const cardValue = clickedCard.dataset.value;


            if (cardType === 'special') {

                if (cardValue === 'octopus') {

                    triggerOctopusEffect(clickedCard);

                    return;

                }

                if (cardValue === 'tornado') {

                    triggerTornadoEffect(clickedCard);

                    return;

                }

            }


            flippedCards.push(clickedCard);

            if (flippedCards.length === 2) {

                isBoardLocked = true;

                checkForMatch();

            }

        }


        function checkForMatch() {

            const [card1, card2] = flippedCards;

            const isMatch = card1.dataset.value === card2.dataset.value;


            if (isMatch) {

                if(card1.dataset.value === 'bomb') {

                    triggerBombEffect(card1, card2);

                    flippedCards = [];

                    return;

                }

                matchedPairs++;

                card1.classList.add('matched');

                card2.classList.add('matched');

                if (matchedPairs === totalPairs) {

                    clearInterval(timerInterval);

                    setTimeout(nextLevel, 1000);

                }

                isBoardLocked = false;

            } else {

                setTimeout(() => {

                    card1.classList.remove('flipped');

                    card2.classList.remove('flipped');

                    isBoardLocked = false;

                }, 1200);

            }

            flippedCards = [];

        }

        function triggerTornadoEffect(card) {

            isBoardLocked = true;

            showTrapMessage("¡Cuidado! Este tornado revuelca todas tus cartas... ¡Ñaca ñaca!");

            const unflippedCards = allCards.filter(c => !c.querySelector('.card').classList.contains('matched') && !c.querySelector('.card').classList.contains('flipped'));

            

            if (unflippedCards.length < 4) { 

                isBoardLocked = false;

                return;

            }


            const cardsToShuffle = shuffle(unflippedCards).slice(0, 4);

            const contents = cardsToShuffle.map(c => c.innerHTML);

            const shuffledContents = shuffle(contents);

                    isBoardLocked = false;

                }, 500);

        }

        function calculateTimeForLevel(currentLevel) {

            const pairs = currentLevel + 1;

            let time = 15;

            time += pairs * 6;


            if (currentLevel >= 4) time += 7;

            if (currentLevel >= 6) time += 12;

            if (currentLevel >= 8) time += 10;


            return time;

        }

            createBoard();

        }


        function nextLevel() {

            if (level >= MAX_LEVEL) {

                showVictoryScreen();

            } else {

                level++;

                startGame();

            }

        }

        

        function startTimer() {

            clearInterval(timerInterval);

            timerInterval = setInterval(() => {

                timeLeft--;

                timerDisplay.textContent = timeLeft;

                if (timeLeft <= 0) {

                    clearInterval(timerInterval);

                    showGameOverScreen();

                }

            }, 1000);

        }


        function showGameOverScreen(title = "¡Tiempo agotado!", message = "Blenderito está decepcionado de ti :(") {

            loseSound();

            isGameActive = false;

            gameOverTitle.textContent = title;

            gameOverMessage.textContent = message;

            gameOverScreen.classList.remove('hidden');

            gameOverScreen.classList.add('flex');

        }


        function showVictoryScreen() {

            winSound();

            isGameActive = false;

            createConfetti();

            victoryScreen.classList.remove('hidden');

            victoryScreen.classList.add('flex');

        }

        

        function createConfetti() {

            const container = document.getElementById('confetti-container');

            if(!container) return;

            container.innerHTML = '';

            const colors = ['#f44336', '#e91e63', '#9c27b0', '#2196f3', '#4caf50', '#ffeb3b'];

            for (let i = 0; i < 100; i++) {

                const c = document.createElement('div');

                c.classList.add('confetti');

                c.style.left = `${Math.random()*100}vw`; c.style.backgroundColor=colors[Math.floor(Math.random()*colors.length)]; c.style.animationDelay=`${Math.random()*2}s`; c.style.transform=`scale(${Math.random()*0.5+0.5})`;

                container.appendChild(c);

            }

        }
